type nat =
  | Z
  | S of nat

type list =
  | Nil
  | Cons of nat * list

type bool =
  | True
  | False

let rec add (n1:nat) (n2:nat) : nat =
  match n1 with
  | Z u -> n2
  | S n3 -> S (add n3 n2)
;; 
let rec double (n1:nat) : nat  |*| (0,0,1)=
  add n1 n1
;; 
let rec mult (n1:nat) (n2:nat) : nat  =
  match n1 with
  | Z u -> Z
  | S n3 -> add n2 (mult n3 n2)
;; 

let rec square (n1:nat)  : nat |*| (0,0,1)  =
  mult n1 n1
;; 

let rec div2 (n1:nat) : nat = 
  match n1 with
  | Z u -> Z
  | S n2 -> match n2 with 
      | Z u -> Z
      | S n3 -> S (div2 n3)
;;

let rec even (n:nat) : bool =
    match n with
      | Z u  -> True
      | S m  -> match (even m) with
            | True u -> False
            | False u -> True
;;
